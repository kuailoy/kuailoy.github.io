                    # 类的定义和修饰符

                    修饰符 用来修饰类的属性或方法

                    public, private, protected

                    class Person {
  public name: string;  // 公有属性, 可以在实例上访问
  private age: number;  // 私有属性, 只能在类自身内访问
  public constructor(name: string, age: number) {
    this.name = name;
    this.age = age;
  }

  protected getAge(): number {
    return this.age;
  }
}

class Woman extends Person {
  constructor(name: string, age: number) {
    super(name, age)
    console.log(super.age);  // error: 只能访问基类被标注为'public'或'protected'的属性
    console.log(super.getAge()); // 18
  }
}
new Woman('Mary', 18);

                    readonly

                    class Person {
  public readonly name: string; // readonly和其他修饰符并用时，放在后面
  readonly age: number;
  constructor(name: string, age: number) {
    this.name = name;
    this.age = age
  }
}
const p = new Person('Jack', 20);
p.name = 'Tom';  // 不能给属性'name'赋值，因为它是一个只读属性

                    # 参数属性

                    在上面的例子中，使用类的实例属性时，都是先在头部声明，然后在构造函数中赋值

                    可以直接在构造函数的参数前加上属性修饰符，即参数属性， 此时，不必在显式地声明该属性，如下

                    class Person {
  constructor(public name: string) { // 参数属性，使用了public修饰符
    this.name = name;
  }
}

                    # 静态属性

                    class Person {
  public static age: number = 18;  // 声明静态属性
  constructor() {}
}
console.log(Person.age); // 18

                    # 可选类属性

                    使用?来标记可选属性

                    class Person {
  public name: string;
  public age?: number;
  constructor(name: string, age: number = 18) {
    this.name = name;
    this.age = age;
  }
}
let p = new Person('Bruce');
console.log(p.age);  // 18

                    # 存取器

                    class Person {
  private _name: string = '';
  constructor() {}

  get name() {
    return this._name;
  }
  set name(value) {
    console.log(`setter: ${value}`);
    this._name = value;
  }
}
const p = new Person();
p.name = 'Jordan';   // setter: Jordan
console.log(p.name); // Jordan

                    # 抽象类

                    abstract用于定义抽象类和其中的抽象方法

                    抽象类不允许被实例化

                    abstract class Person {}
const p = new Person();  // 不能创建抽象类的实例

                    抽象类中的抽象方法必须被子类实现

                    abstract class Person {
  public name: string;
  public constructor(name: string) {
    this.name = name;
  }
  public abstract sayHi(): void;
}

class Woman extends Person {} // 非抽象类'Woman'没有实现'Perosn'抽象成员'sayHi'

                    正确使用

                    abstract class Person {
  public name: string;
  public constructor(name: string) {
    this.name = name;
  }
  public abstract sayHi(): void;
}
class Woman extends Person {
  public sayHi() {
    console.log(`Hi, I'm ${this.name}`)
  }
}
const p = new Woman('Jane');
p.sayHi();  // Hi, I'm Jane

                    # 类实现接口

                    使用接口定义一个类包含的属性

                    interface IFruit {
  type: string;
}
class Fruit implements IFruit {
  // error: 'Fruit'类对接口'IFruit'的实现不正确, 缺少'type'属性
}

                    正确用法

                    interface IFruit {
  type: string;
}
// 只检验是否实现了type属性，多余属性则不检验
class Fruit implements IFruit {
  public type: string;
  public color: string
  constructor(type: string, color: string) {
    this.type = type;
    this.color = color;
  }
}

                    # 接口继承类

                    接口可以继承类的成员及类型

                    class Fruit {
  public type: string;
  constructor(type: string) {
    this.type = type;
  }
}
interface IFruit extends Fruit {}

                    当接口继承的了private和protected`修饰的属性时，该接口只能被该类或其子类实现

                    class Fruit {
  protected type: string;
  constructor(type: string) {
    this.type = type;
  }
}
interface IFruit extends Fruit {}

class Dog implements IFruit {
  // error: 属性'type'是受保护类型，然而'Dog'类并非继承自'Fruit'类
  type: string = 'dog'；
}
//正确用法
class Apple extends Fruit implements IFruit {
  type: string = 'apple'
}

                    # 泛型与类类型的结合

                    下面的函数接收一个类作为参数，返回该类的实例

                    function create<T>(c: {new(): T}): T {
  return new c();
}
class Demo {}
create(Demo);

                    简单分析下，参数类型为{new(): T}, 即类类型，泛型T则表示该类的构造函数，与实例的类型一致，表示返回值也是该类实例的类型
