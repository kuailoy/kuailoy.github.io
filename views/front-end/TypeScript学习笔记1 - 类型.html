    TypeScript学习笔记1 - 类型

    # 类型声明语法

    let num: number;
let str: string = 'string';
num = 'abc'; // error 不能将类型"abc"分配给类型"number"

    # JS 原有类型

    TypeScript 中包含八个 JavaScript 的原有的类型：string, number, boolean, undefined, null, symbol, array, object

    其中基本类型不再赘述，简单说下这几种：

    # 数组类型语法

    // 声明一个number类型的数组
let arr1: number[] = [1, 2, 3]; // 推荐写法
let arr2: Array<number> = [1, 2, 3];
let arr3: number | string[] = [1, '2', 3]; // 联合类型

    # object 类型

    // object类型的错误用法
let obj: object = { name: 'Jack' };
console.log(obj.name); // error name属性不存在

// object类型的正确用法
function getKeys(target: object) {
  return Object.keys(target);
}
getKeys({ name: 'Jack' }); // ['name']
getKeys(123); // error 类型"123"的参数不能赋给类型"object"的参数

    undefined / null 类型可以赋值给任意类型

    当设置"strictNullChecks": true时，undefined / null 只能赋值给他们自身的类型和 void 类型（实际上，null 不能赋值给 void 类型）

    # TS 补充的类型

    # 元组（Tuple）

    元组可以看作是已知元素数量(length)，和每个位置(index)确切的元素类型(type)的数组

    let tuple: [string, number];
tuple = ['a', 1];
tuple = [1, 'a']; // error
tuple = ['a', 1, 2]; // error
tuple[1] = 2;
tuple[1] = 'b'; //error

    # 枚举（enum）

    给一组相关的数值赋予名字

    // 默认值分别为0，1，2，3
enum directions {
  TOP, // 0
  RIGHT, // 1
  DOWN, // 2
  LEFT, // 3
}

// 给初始值情况下，后面递增
enum fruits {
  APPLE = 1,
  ORANGE, //2
  BANANA, //3
}

// 直接赋值
enum animals {
  CAT = 2,
  DOG = 4,
  PIG = 6,
}

console.log(fruits.BANANA); // 1
console.log(animals[4]); // 'DOG'

    # any

    有时候我们在声明变量时并不能确切知道它的类型， 这时候就可以用到any

    // any类型变量
let a: any;
a = 123;
a = 'hello';
a = true;
// any类型的数组
let arr: any[] = [123, 'hello', true];

    注意，不要滥用any，否则typescript就变成了anyscript, 更安全的做法是使用unknown类型

    # void

    void表示没有任何类型，当一个函数没有返回值时可以使用void

    function consoleValue(value: string): void {
  console.log(value);
}

    void 类型只能赋值为undefined和null(严格模式下只能赋值为undefined)

    # never

    never值表示永不存在的值的类型，例如总会抛出异常或永远不会有返回值的函数

    const throwError = (msg: string): never => {
  throw new Error(msg);
};
const infiniteFunc = (): never => {
  while (true) {}
};

    never与void的区别：void 是定义函数是没有给它返回值，而never是根本不会有返回值，比如一个死循环

    never可以赋值给任何类型，任何类型都不能赋值给 never（包括 any）

    let err: string = throwError('error'); // pass
infiniteFunc = 123; // error

    # unknown

    let a: any;
a.name;
a.toFixed();
a.length;

    将一个变量指定为any类型时， 以上均可编译通过

    不同于any, 当你给它类型断言之前是无法调用任何方法的，所以说 unknown 相对于 any 是更安全的

    # 交叉类型

    function merge<T, U>(obj1: T, obj2: U): T & U {
  // 返回值result是交叉类型，该类型的值同时拥有obj1和obj2的属性
  let result = <T & U>{};
  result = Object.assign(obj1, obj2);
  return result;
}

const a = {
  name: 'Jack',
};
const b = {
  age: 18,
};
const obj = merge(a, b); // 这里的返回值就是交叉类型，必须同时具有a和b的属性
obj.gender = 'male'; // error 类型 "{name: string} & {age: number}"上不存在属性"gender"

    # 联合类型

    function getLength(value: string | number): number {
  // 这里的参数value使用了联合类型，value值符合string或者number其中之一即可
  if (typeof value === 'string') {
    return value.length;
  } else {
    return value.toString().length;
  }
}
console.log(getLength('abc'));
console.log(getLength(123));
